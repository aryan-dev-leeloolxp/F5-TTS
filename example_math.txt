public class Complex {
    private double real, imag;
    public Complex(double r, double i) {
        this.real = r;    // Real part
        this.imag = i;    // Imaginary part
    }
    public Complex add(Complex c) {
        return new Complex(
            real + c.real,    // Add reals
            imag + c.imag     // Add imaginaries
        );
    }
}
